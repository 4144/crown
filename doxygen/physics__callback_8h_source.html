59  void onContact(const PxContactPairHeader& pair_header, const PxContactPair* pairs, PxU32 num_pairs)
60  {
61  // Do not report contact if either actor0 or actor1 or both have been deleted
62  if (pair_header.flags & PxContactPairHeaderFlag::eDELETED_ACTOR_0 ||
63  pair_header.flags & PxContactPairHeaderFlag::eDELETED_ACTOR_1) return;
64 
65  for (PxU32 pp = 0; pp < num_pairs; pp++)
66  {
67  const PxContactPair& cp = pairs[pp];
68 
69  // We are only interested in touch found or lost
70  if ((cp.events & PxPairFlag::eNOTIFY_TOUCH_FOUND) == (PxPairFlag::Enum)0 &&
71  (cp.events & PxPairFlag::eNOTIFY_TOUCH_LOST) == (PxPairFlag::Enum)0) continue;
72 
73  // Skip if either shape0 or shape1 or both have been deleted
74  if (cp.flags & PxContactPairFlag::eDELETED_SHAPE_0 ||
75  cp.flags & PxContactPairFlag::eDELETED_SHAPE_1) continue;
76 
77  PxContactPairPoint points[8];
78  const PxU32 num_points = cp.extractContacts(points, 8);
79 
80  PxVec3 where(0, 0, 0);
81  PxVec3 normal(0, 0, 0);
82  for (PxU32 i = 0; i < num_points; i++)
83  {
84  where = points[i].position;
85  normal = points[i].normal;
86  }
87 
88  post_collision_event((Actor*) pair_header.actors[0]->userData,
89  (Actor*) pair_header.actors[1]->userData,
90  vector3(where.x, where.y, where.z),
91  vector3(normal.x, normal.y, normal.z),
92  (cp.events & PxPairFlag::eNOTIFY_TOUCH_FOUND) ?
93  physics_world::CollisionEvent::BEGIN_TOUCH :
94  physics_world::CollisionEvent::END_TOUCH);
95  }
96  }

98  void onTrigger(PxTriggerPair* pairs, PxU32 count)
99  {
100  for (PxU32 pp = 0; pp < count; pp++)
101  {
102  const PxTriggerPair& tp = pairs[pp];
103 
104  // Do not report event if either trigger ot other shape or both have been deleted
105  if (tp.flags & PxTriggerPairFlag::eDELETED_SHAPE_TRIGGER ||
106  tp.flags & PxTriggerPairFlag::eDELETED_SHAPE_OTHER) continue;
107 
108  post_trigger_event((Actor*)tp.triggerActor->userData,
109  (Actor*)tp.otherActor->userData,
110  (tp.status & PxPairFlag::eNOTIFY_TOUCH_FOUND ?
111  physics_world::TriggerEvent::BEGIN_TOUCH : physics_world::TriggerEvent::END_TOUCH));
112  }
