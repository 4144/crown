45 inline Color4 from_rgba(uint32_t rgba)
46 {
47  Color4 c;
48  c.x = 1.0f/255.0f * ((rgba & 0xff000000) >> 24);
49  c.y = 1.0f/255.0f * ((rgba & 0x00ff0000) >> 16);
50  c.z = 1.0f/255.0f * ((rgba & 0x0000ff00) >> 8);
51  c.w = 1.0f/255.0f * ((rgba & 0x000000ff) >> 0);
52  return c;
53 }

57 {
58  uint32_t rgba;

60  rgba |= (uint32_t)(255.0f * c.y) << 16;
