98  CE_UNUSED(err);
99  return info.st_mtime;
100 #elif CROWN_PLATFORM_WINDOWS
101  HANDLE hfile = CreateFile(path
102  , GENERIC_READ

104  , NULL
105  , OPEN_EXISTING
106  , 0
107  , NULL
108  );
109  CE_ASSERT(hfile != INVALID_HANDLE_VALUE, "CreateFile: GetLastError = %d", GetLastError());
110  FILETIME ftwrite;

115  return (uint64_t)((uint64_t(ftwrite.dwHighDateTime) << 32) | ftwrite.dwLowDateTime);

185  inline void list_files(const char* path, Vector<DynamicString>& files)
186  {
187 #if CROWN_PLATFORM_POSIX
188  DIR *dir;
189  struct dirent *entry;
190 
191  if (!(dir = opendir(path)))
192  return;
193 
194  while ((entry = readdir(dir)))
195  {
196  const char* dname = entry->d_name;
197 
198  if (!strcmp(dname, ".") || !strcmp(dname, ".."))
199  continue;
200 
201  TempAllocator512 ta;
202  DynamicString fname(dname, ta);
203  vector::push_back(files, fname);
204  }
205 
206  closedir(dir);
207 #elif CROWN_PLATFORM_WINDOWS
208  TempAllocator1024 ta;
209  DynamicString cur_path(path, ta);
210  cur_path += "\\*";
211 
212  WIN32_FIND_DATA ffd;
213  HANDLE file = FindFirstFile(cur_path.c_str(), &ffd);
214  if (file == INVALID_HANDLE_VALUE)
215  return;
216 
217  do
218  {
219  const char* fname = ffd.cFileName;
220 
221  if (!strcmp(fname, ".") || !strcmp(fname, ".."))
222  continue;
223 
224  TempAllocator512 ta;
225  DynamicString filename(fname, ta);
226  vector::push_back(files, filename);
227  }
228  while (FindNextFile(file, &ffd) != 0);
229 
230  FindClose(file);
231 #endif
232  }

314  inline int execute_process(const char* path, const char* args, StringStream& output)
315  {
316 #if CROWN_PLATFORM_POSIX
317  using namespace string_stream;
318 
319  TempAllocator512 ta;
320  DynamicString cmd(path, ta);
321  cmd += " 2>&1 ";
322  cmd += args;
323  FILE* file = popen(cmd.c_str(), "r");
324 
325  char buf[1024];
326  while (fgets(buf, sizeof(buf), file) != NULL)
327  output << buf;
328 
329  return pclose(file);
330 #elif CROWN_PLATFORM_WINDOWS
331  STARTUPINFO info;
332  memset(&info, 0, sizeof(info));
333  info.cb = sizeof(info);
334 
335  PROCESS_INFORMATION process;
336  memset(&process, 0, sizeof(process));
337 
338  int err = CreateProcess(path, (LPSTR)args, NULL, NULL, TRUE, 0, NULL, NULL, &info, &process);
339  CE_ASSERT(err != 0, "CreateProcess: GetLastError = %d", GetLastError());
340  CE_UNUSED(err);
341 
342  DWORD exitcode = 1;
343  ::WaitForSingleObject(process.hProcess, INFINITE);
344  GetExitCodeProcess(process.hProcess, &exitcode);
345  CloseHandle(process.hProcess);
346  CloseHandle(process.hThread);
347  return (int)exitcode;
348 #endif
