94 template <uint32_t SIZE>
95 inline void* TempAllocator<SIZE>::allocate(uint32_t size, uint32_t align)
96 {
97  _cur = (char*) memory::align_top(_cur, align);
98 
99  if (size > uint32_t(_end - _cur))
100  {
101  uint32_t to_allocate = sizeof(void*) + size + align;
102 
103  if (to_allocate < _chunk_size)
104  {
105  to_allocate = _chunk_size;
106  }
107 
108  _chunk_size *= 2;
109 
110  void *p = _backing.allocate(to_allocate);
111  *(void **)_begin = p;
112  _cur = _begin = (char*) p;
113  _end = _begin + to_allocate;
114  *(void**) _begin = 0;
115  _cur += sizeof(void*);
116  memory::align_top(p, align);
117  }
118 
119  void *result = _cur;
120  _cur += size;
121  return result;
122 }
