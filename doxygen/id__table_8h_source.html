60  template <uint32_t MAX>
61  inline Id create(IdTable<MAX>& a)
62  {
63  // Obtain a new id
64  Id id;
65  id.id = a._next_id++;
66 
67  // Recycle slot if there are any
68  if (a._freelist != INVALID_ID)
69  {
70  id.index = a._freelist;
71  a._freelist = a._ids[a._freelist].index;
72  }
73  else
74  {
75  id.index = a._size;
76  }
77 
78  a._ids[id.index] = id;
79  a._size++;
80 
81  return id;
82  }
