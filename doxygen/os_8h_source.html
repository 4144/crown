135  inline bool is_directory(const char* path)
136  {
137 #if CROWN_PLATFORM_POSIX
138  struct stat info;
139  memset(&info, 0, sizeof(info));
140  int err = lstat(path, &info);
141  CE_ASSERT(err == 0, "lstat: errno = %d", errno);
142  CE_UNUSED(err);
143  return ((S_ISDIR(info.st_mode) == 1) && (S_ISLNK(info.st_mode) == 0));
144 #elif CROWN_PLATFORM_WINDOWS
145  DWORD fattr = GetFileAttributes(path);
146  return (fattr != INVALID_FILE_ATTRIBUTES && (fattr & FILE_ATTRIBUTE_DIRECTORY) != 0);
147 #endif

167  inline u64 mtime(const char* path)
168  {
169 #if CROWN_PLATFORM_POSIX
170  struct stat info;
171  memset(&info, 0, sizeof(info));
172  int err = lstat(path, &info);
173  CE_ASSERT(err == 0, "lstat: errno = %d", errno);
174  CE_UNUSED(err);
175  return info.st_mtime;
176 #elif CROWN_PLATFORM_WINDOWS
177  HANDLE hfile = CreateFile(path
178  , GENERIC_READ
179  , FILE_SHARE_READ
180  , NULL
181  , OPEN_EXISTING
182  , 0
183  , NULL
184  );
185  CE_ASSERT(hfile != INVALID_HANDLE_VALUE, "CreateFile: GetLastError = %d", GetLastError());
186  FILETIME ftwrite;
187  BOOL err = GetFileTime(hfile, NULL, NULL, &ftwrite);
188  CE_ASSERT(err != 0, "GetFileTime: GetLastError = %d", GetLastError());
189  CE_UNUSED(err);
190  CloseHandle(hfile);
191  return (u64)((u64(ftwrite.dwHighDateTime) << 32) | ftwrite.dwLowDateTime);
192 #endif
193  }
