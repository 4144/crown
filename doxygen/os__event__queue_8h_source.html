294  bool push_event(const OsEvent& ev)
295  {
296  int cur_tail = _tail.load();
297  int next_tail = increment(cur_tail);
298  if(next_tail != _head.load())
299  {
300  _queue[cur_tail] = ev;
301  _tail.store(next_tail);
302  return true;
303  }
304 
305  return false;
306  }
307 
308  bool pop_event(OsEvent& ev)
309  {
310  const int cur_head = _head.load();
311  if(cur_head == _tail.load()) return false;
312 
313  ev = _queue[cur_head];
314  _head.store(increment(cur_head));
315  return true;
316  }
317 
318  int increment(int idx) const
319  {
320  return (idx + 1) % MAX_OS_EVENTS;
321  }
