104  template<typename T> FindResult find(const Hash<T>& h, u64 key)
105  {
106  FindResult fr;
107  fr.hash_i = END_OF_LIST;
108  fr.data_prev = END_OF_LIST;
109  fr.data_i = END_OF_LIST;
110 
111  if (array::size(h._hash) == 0)
112  return fr;
113 
114  fr.hash_i = key % array::size(h._hash);
115  fr.data_i = h._hash[fr.hash_i];
116  while (fr.data_i != END_OF_LIST) {
117  if (h._data[fr.data_i].key == key)
118  return fr;
119  fr.data_prev = fr.data_i;
120  fr.data_i = h._data[fr.data_i].next;
121  }
122  return fr;
123  }

146  template<typename T> void erase(Hash<T>& h, const FindResult &fr)
147  {
148  if (fr.data_prev == END_OF_LIST)
149  h._hash[fr.hash_i] = h._data[fr.data_i].next;
150  else
151  h._data[fr.data_prev].next = h._data[fr.data_i].next;
152 
153  if (fr.data_i == array::size(h._data) - 1) {
154  array::pop_back(h._data);
155  return;
156  }
157 
158  h._data[fr.data_i] = h._data[array::size(h._data) - 1];
159  FindResult last = find(h, h._data[fr.data_i].key);
160 
161  if (last.data_prev != END_OF_LIST)
162  h._data[last.data_prev].next = fr.data_i;
163  else
164  h._hash[last.hash_i] = fr.data_i;
165  }

172  template<typename T> u32 find_or_make(Hash<T>& h, u64 key)
173  {
174  const FindResult fr = find(h, key);
175  if (fr.data_i != END_OF_LIST)
176  return fr.data_i;

179  if (fr.data_prev == END_OF_LIST)

207  template<typename T> void rehash(Hash<T>& h, u32 new_size)
208  {
209  Hash<T> nh(*h._hash._allocator);
210  array::resize(nh._hash, new_size);
211  array::reserve(nh._data, array::size(h._data));
212  for (u32 i=0; i<new_size; ++i)
213  nh._hash[i] = END_OF_LIST;
214  for (u32 i=0; i<array::size(h._data); ++i) {
215  const typename Hash<T>::Entry &e = h._data[i];
216  multi_hash::insert(nh, e.key, e.value);
217  }
218 
219  Hash<T> empty(*h._hash._allocator);
220  h.~Hash<T>();
221  memcpy(&h, &nh, sizeof(Hash<T>));
222  memcpy(&nh, &empty, sizeof(Hash<T>));
223  }
